Array 객체의 메소드 활용

배열 a = 황, 김, 이

배열 b = 박


let a = new Array("황", "김", "이");

let b = new Array("박");

let c

concat(arr)

현재 배열에 배열 arr의 원소들을 덧붙여 만든 새 배열 리턴, 현재 배열은 변하지 않음.


let a = new Array("황", "김", "이");

let b = new Array("박");

let c

c = a.concat(b) ----- // c는 a와 b를 연결한 새 배열

c = 황,김,이,박

join([separator])

배열의 모든 원소를 연결하여 하나의 문자열로 만들어 리턴

각 원소 사이에 separator로 지정된 문자열 삽입 separator가 생략되면 "," 이 사용


let a = new Array("황", "김", "이");

let b = new Array("박");

let c

c = a.join("##") ----- // c는 배열a 각 원소 사이에, ##을 삽입한 형태의 문자열

c = 황##김##이

reverse()

원소를 역순으로 재배열하고, 이를 복사한 새로운 배열 리턴


let a = new Array("황", "김", "이");

let b = new Array("박");

let c

c = a.reverse() ----- // c는 배열a를 역순으로 재배열한, 새로운 배열

c = 이,김,황

a = 이,김,황 ----- // a자체를 새로운 배열로 만들었기 때문에 a도 변함

slice(idxA, idxB)

두 인덱스 idxA 와 idxB 사이의 원소들로 구성된 새 배열 리턴

idxB 의 원소는 포함되지 않음, idxB가 생략되면 끝 원소까지 포함


let a = new Array("황", "김", "이");

let b = new Array("박");

let c

c = a.slice(1, 2) ----- // c는 배열a의 1번원소와 2번원소 사이의 범위를 복사해 새로운 배열 리턴

c = 김 ----- // "김" 부터 "이" 까지 복사, idxB는 포함하지 않기 때문에 "김"만 출력됨


c = a.slice(1) ----- // idxB 의 원소는 포함되지 않음, idxB가 생략되면 끝 원소까지 포함

c = 김,이 ----- // "김" 부터 복사, idxB를 생략해서 "이"까지 출력됨



sort()

문자열 배열의 경우 문자열을 사전 순으로 정렬하고, 이를 복사한 새 배열 리턴.


let a = new Array("황", "김", "이");

let b = new Array("박");

let c

c = a.sort() ----- // c는 배열a를 사전순으로 정렬해 새로운 배열 리턴

c = 김,이,황 ----- // 사전 순으로 출력

a = 김,이,황 ----- // a자체를 새로운 배열로 만들었기 때문에 a도 변함

toString()

원소들을 모두 연결하여 하나의 문자열로 만들어 리턴, 원소 사이에 "," 삽입


let a = new Array("황", "김", "이");

let b = new Array("박");

let c

c = a.toString() ----- // toString은 원소 사이에 ","를 넣어 연결된 문자열 생성

c = 황,김,이
