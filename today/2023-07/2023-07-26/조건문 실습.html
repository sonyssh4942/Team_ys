조건문 실습

prompt()를 이용하여 커피 이름을 입력받고 switch문을 이용하여 가격을 출력하라

switch 문으로 커피 주문

var coffee = prompt("무슨 커피를 드릴까요?", ""); ----- // coffee 변수에 prompt 다이얼로그로 커피이름을 입력받음
var price = 0; ----- // price 변수에 커피값을 대입할 것 이기 때문에 0원으로 초기화

switch(coffee){ ----- // switch문으로 커피이름과 가격을 지정
case "아메리카노": price = 1000;
break;
case "카페라떼": price = 2000;
break;
case "에스프레소": price = 3000;
break;
default : document.write(coffee + "는 없습니다."); ----- // 이름을 잘못 입력했을때 출력될 문구
}


// 이름을 맞게 입력했다면, 이름과 가격을 출력
if(price != 0){ -----// 이름을 맞게 입력했다면, 가격이 0원이 아니기 때문에 조건식에 0원이 아닐경우를 가정
document.write("주문하신 커피는 " + coffee + " 입니다.")
document.write("커피의 가격은 " + price + "원 입니다.")

}
